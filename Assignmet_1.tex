\section*{\textbf{Problem}}
\textbf{\textsl{1. The co-ordinates of points A,B are (r_1,$\theta_1$) , (r_2,$\theta_2$) referred to O as pole. The internal bisector of angle AOB meets the line AB in D. Find the co-ordinates of D.}}
\noindent
\section*{\textbf{Solution}}
\textsl{Let the co-ordinates of point D be (r,\theta).}

\textsl{}

Now , \theta = \dfrac{\theta_1+\theta_2}{2}

\textsl{}

Since , Area of $\triangle$ AOD + Area of $\triangle$ DOB = Area of $\triangle$ AOB

\textsl{}

$\therefore $ \dfrac{1}{2}r_1 r sin(\dfrac{\theta}{2}) + \dfrac{1}{2}r_2 r  sin(\dfrac{\theta}{2}) = \dfrac{1}{2}r_1 r_2 sin(\theta)
\begin{flushright}
.....(i)
\end{flushright}

\textsl{}

\textsl{Now , R.H.S = \dfrac{1}{2}r_1 r_2 sin(\theta)}

\textsl{}

\textsl{ = \dfrac{1}{2}r_1 r_2 2sin(\dfrac{\theta}{2})cos(\dfrac{\theta}{2})}
\begin{flushright}
.....(ii)
\end{flushright}

\textsl{}

\textsl{On Comparing (i) and (ii)}

\textsl{}

\textsl{$\dfrac{1}{2}$ r (r_1 + r_2) = r_1 r_2 \cos(\dfrac{\theta}{2})}

\textsl{}

\textsl{$\therefore$ r = \dfrac{2r_1 r_2}{r_1+r_2}cos(\dfrac{\theta}{2})}

\textsl{}

\textsl{Now , D has co-ordinates as (r,\theta)}

\textsl{}

\textsl{$\therefore$ Co-ordinates of D are (\dfrac{2r_1r_2}{r_1+r_2}cos(\dfrac{\theta}{2}}),\dfrac{\theta_1+\theta_2}{2})}
